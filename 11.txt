git restore  文件名  用暂存区文件恢复工作区文件区

git reset  
           --mixed  默认 只恢复暂存区和本地仓库  工作区不改变
           --soft       只把本地仓库回退， 暂存区和工作区不变
           --hard      暂存区、本地仓库、和工作区都回退

git reset  --hard 哈希码
git reset  --hard HEAD^
